// 2023. 10. 2 월										(5주 1일)
// 10월 25일(8주 2일) - 중간시험
//---------------------------------------------------------------------------
// 정렬(sort)
//---------------------------------------------------------------------------

#include <iostream>
#include <format>
#include <random>
#include "save.h"
using namespace std;

default_random_engine dre;
uniform_int_distribution uid{ 1, 999 };

// [문제] int 값 1'000개를 저장할 수 있는 배열을 만들어라
// 각 int값을 1부터 999까지 값 중 랜덤값으로 만들어라
// 배열을 오름차순으로 정렬하라
// 전체 1000개 값을 화면에 출력하라

//--------
int main()
//--------
{
	int num[1000];		// 4KB

	for (int i = 0; i < 1000; ++i) {
		num[i] = uid(dre);
	}

int 정렬방법(const void* p, const void* q) {
	int a = *(int*)p;
	int b = *(int*)q;

	// 오름차순
	if (a > b) {
		return 1;
	}
	else if (a < b) {
		return -1;
	}

	return 0;

}
